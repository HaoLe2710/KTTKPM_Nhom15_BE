Below is a **complete tasklist** for you to implement **Phase 1** and **Phase 2** for extending your semantic data catalog, following the current state of your database: currently, the catalog only has `product_types`, `products`, `variants`, `options`, `option_values`, `variant_options`, `media`, while the search side already has full projection/metadata.
The goal of this tasklist is:
* Make the `catalog` the **source of truth** for semantic data.
* so that the search function no longer has to "guess" facets from poor data.
* Upgrade the projection builder to build facets and search documents more accurately.
* Follow each phase to avoid disrupting the ongoing flow.

# PHASE 1 — Minimum Semantic Source Model
#1. Phase 1 Objectives
After Phase 1, your system should achieve the following:
* The product can be branded.
* The product may contain ingredients.
* The product can be fitted with various skin types.
* Product may have concerns
* Products can be tagged
Projector builder can build facets:
* `ingredient`
* `skin_type`
* `concern`
* `tag`
* retain the attribute from option values
* Search preview/admin search status is starting to become more meaningful.
---
#2. DDL Migration — Phase 1
I recommend splitting it into **2 migrations**.
## Migration P1.1 — master tables + product columns
```sql
-- =========================
-- PHASE 1.1 - MASTER DATA
-- =========================
CREATE TABLE brands
id VARCHAR(255) PRIMARY KEY,
code VARCHAR(100) UNIQUE NOT NULL,
name VARCHAR(255) NOT NULL,
slug VARCHAR(255) UNIQUE NOT NULL,
Description TEXT,
logo_url VARCHAR(500),
is_active BOOLEAN DEFAULT TRUE NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at TIMESTAMP
);
CREATE TABLE ingredients
id VARCHAR(255) PRIMARY KEY,
code VARCHAR(100) UNIQUE NOT NULL,
name VARCHAR(255) NOT NULL,
normalized_name VARCHAR(255),
inci_name VARCHAR(255),
Description TEXT,
is_active BOOLEAN DEFAULT TRUE NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at TIMESTAMP
);
CREATE TABLE skin_types (
id VARCHAR(255) PRIMARY KEY,
code VARCHAR(100) UNIQUE NOT NULL,
name VARCHAR(255) NOT NULL,
Description TEXT,
is_active BOOLEAN DEFAULT TRUE NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at TIMESTAMP
);
CREATE TABLE concerns
id VARCHAR(255) PRIMARY KEY,
code VARCHAR(100) UNIQUE NOT NULL,
name VARCHAR(255) NOT NULL,
Description TEXT,
is_active BOOLEAN DEFAULT TRUE NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at TIMESTAMP
);
CREATE TABLE tags (
id VARCHAR(255) PRIMARY KEY,
code VARCHAR(100) UNIQUE NOT NULL,
name VARCHAR(255) NOT NULL,
Description TEXT,
is_active BOOLEAN DEFAULT TRUE NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at TIMESTAMP
);
ALTER TABLE products
ADD COLUMN brand_id VARCHAR(255),
ADD COLUMN short_description VARCHAR(500);
ALTER TABLE products
ADD CONSTRAINT fk_products_brand
FOREIGN KEY (brand_id) REFERENCES brands(id);
CREATE INDEX idx_products_brand_id ON products(brand_id);
CREATE INDEX idx_brands_slug ON brands(slug);
CREATE INDEX idx_ingredients_normalized_name ON ingredients(normalized_name);
```

## Migration P1.2 — relation tables
```sql
-- =========================
-- PHASE 1.2 - PRODUCT RELATIONS
-- =========================
CREATE TABLE product_ingredients (
id VARCHAR(255) PRIMARY KEY,
product_id VARCHAR(255) NOT NULL,
ingredient_id VARCHAR(255) NOT NULL,
concentration_label VARCHAR(100),
is_highlighted BOOLEAN DEFAULT FALSE NOT NULL,
sort_order INT DEFAULT 0 NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at TIMESTAMP,
CONSTRAINT fk_product_ingredients_product
FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE,
CONSTRAINT fk_product_ingredients_ingredient
FOREIGN KEY (ingredient_id) REFERENCES ingredients(id) ON DELETE CASCADE,
CONSTRAINT uq_product_ingredient UNIQUE (product_id, ingredient_id)
);
CREATE TABLE product_skin_types (
id VARCHAR(255) PRIMARY KEY,
product_id VARCHAR(255) NOT NULL,
skin_type_id VARCHAR(255) NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at TIMESTAMP,
CONSTRAINT fk_product_skin_types_product
FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE,
CONSTRAINT fk_product_skin_types_skin_type
FOREIGN KEY (skin_type_id) REFERENCES skin_types(id) ON DELETE CASCADE,
CONSTRAINT uq_product_skin_type UNIQUE (product_id, skin_type_id)
);
CREATE TABLE product_concerns (
id VARCHAR(255) PRIMARY KEY,
product_id VARCHAR(255) NOT NULL,
concern_id VARCHAR(255) NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at TIMESTAMP,
CONSTRAINT fk_product_concerns_product
FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE,
CONSTRAINT fk_product_concerns_concern
FOREIGN KEY (concern_id) REFERENCES concerns(id) ON DELETE CASCADE,
CONSTRAINT uq_product_concern UNIQUE (product_id, concern_id)
);
CREATE TABLE product_tags (
id VARCHAR(255) PRIMARY KEY,
product_id VARCHAR(255) NOT NULL,
tag_id VARCHAR(255) NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at TIMESTAMP,
CONSTRAINT fk_product_tags_product
FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE,
CONSTRAINT fk_product_tags_tag
FOREIGN KEY (tag_id) REFERENCES tags(id) ON DELETE CASCADE,
CONSTRAINT uq_product_tag UNIQUE (product_id, tag_id)
);
CREATE INDEX idx_product_ingredients_product_id ON product_ingredients(product_id);
CREATE INDEX idx_product_ingredients_ingredient_id ON product_ingredients(ingredient_id);
CREATE INDEX idx_product_skin_types_product_id ON product_skin_types(product_id);
CREATE INDEX idx_product_skin_types_skin_type_id ON product_skin_types(skin_type_id);
CREATE INDEX idx_product_concerns_product_id ON product_concerns(product_id);
CREATE INDEX idx_product_concerns_concern_id ON product_concerns(concern_id);
CREATE INDEX idx_product_tags_product_id ON product_tags(product_id);
CREATE INDEX idx_product_tags_tag_id ON product_tags(tag_id);
```

#6. DDL Migration — Phase 2
I recommend splitting it into **3 migrations**.
## Migration P2.1 — benefits + collections
```sql
-- =========================
-- PHASE 2.1 - BENEFITS & COLLECTIONS
-- =========================
CREATE TABLE benefits
id VARCHAR(255) PRIMARY KEY,
code VARCHAR(100) UNIQUE NOT NULL,
name VARCHAR(255) NOT NULL,
Description TEXT,
is_active BOOLEAN DEFAULT TRUE NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at TIMESTAMP
);
CREATE TABLE collections
id VARCHAR(255) PRIMARY KEY,
code VARCHAR(100) UNIQUE NOT NULL,
name VARCHAR(255) NOT NULL,
slug VARCHAR(255) UNIQUE NOT NULL,
Description TEXT,
is_active BOOLEAN DEFAULT TRUE NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at TIMESTAMP
);
CREATE TABLE product_benefits (
id VARCHAR(255) PRIMARY KEY,
product_id VARCHAR(255) NOT NULL,
benefit_id VARCHAR(255) NOT NULL,
priority INT DEFAULT 0 NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at TIMESTAMP,
CONSTRAINT fk_product_benefits_product
FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE,
CONSTRAINT fk_product_benefits_benefit
FOREIGN KEY (benefit_id) REFERENCES benefits(id) ON DELETE CASCADE,
CONSTRAINT uq_product_benefit UNIQUE (product_id, benefit_id)
);
CREATE TABLE product_collections (
id VARCHAR(255) PRIMARY KEY,
product_id VARCHAR(255) NOT NULL,
collection_id VARCHAR(255) NOT NULL,
sort_order INT DEFAULT 0 NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at TIMESTAMP,
CONSTRAINT fk_product_collections_product
FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE,
CONSTRAINT fk_product_collections_collection
FOREIGN KEY (collection_id) REFERENCES collections(id) ON DELETE CASCADE,
CONSTRAINT uq_product_collection UNIQUE (product_id, collection_id)
);
CREATE INDEX idx_product_benefits_product_id ON product_benefits(product_id);
CREATE INDEX idx_product_benefits_benefit_id ON product_benefits(benefit_id);
CREATE INDEX idx_product_collections_product_id ON product_collections(product_id);
CREATE INDEX idx_product_collections_collection_id ON product_collections(collection_id);
```

## Migration P2.2 — attributes
```sql
-- =========================
-- PHASE 2.2 - ATTRIBUTES
-- =========================
CREATE TABLE attributes (
id VARCHAR(255) PRIMARY KEY,
code VARCHAR(100) UNIQUE NOT NULL,
name VARCHAR(255) NOT NULL,
data_type VARCHAR(50) NOT NULL, -- TEXT, NUMBER, BOOLEAN
unit VARCHAR(50),
is_filterable BOOLEAN DEFAULT TRUE NOT NULL,
is_searchable BOOLEAN DEFAULT TRUE NOT NULL,
is_active BOOLEAN DEFAULT TRUE NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at TIMESTAMP
);
CREATE TABLE product_attribute_values (
id VARCHAR(255) PRIMARY KEY,
product_id VARCHAR(255) NOT NULL,
attribute_id VARCHAR(255) NOT NULL,
value_text VARCHAR(500),
value_number NUMERIC(18,2),
value_boolean BOOLEAN,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
updated_at TIMESTAMP,
CONSTRAINT fk_product_attribute_values_product
FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE,
CONSTRAINT fk_product_attribute_values_attribute
FOREIGN KEY (attribute_id) REFERENCES attributes(id),
CONSTRAINT uq_product_attribute UNIQUE (product_id, attribute_id)
);
CREATE INDEX idx_product_attribute_values_product_id ON product_attribute_values(product_id);
CREATE INDEX idx_product_attribute_values_attribute_id ON product_attribute_values(attribute_id);
CREATE INDEX idx_product_attribute_values_value_text ON product_attribute_values(value_text);
CREATE INDEX idx_product_attribute_values_value_number ON product_attribute_values(value_number);
CREATE INDEX idx_product_attribute_values_value_boolean ON product_attribute_values(value_boolean);
```
